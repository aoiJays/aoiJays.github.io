class MyDataet(Dataset):

    def __init__(self, root_dir, label):
        
        # è‡ªå®šä¹‰ æ€ä¹ˆæ–¹ä¾¿æ€ä¹ˆæ¥
        self.root_dir = root_dir # è®°å½•æ•°æ®åœ°å€ä»¥åŠå¯¹åº”çš„æ ‡ç­¾
        self.label = label
        self.imglist = os.listdir(self.root_dir) # ä»¥åˆ—è¡¨å½¢å¼å±•ç¤ºæ–‡ä»¶å¤¹å†…æ‰€æœ‰æ–‡ä»¶
    
    def __getitem__(self, idx):

        img_path = os.path.join( self.root_dir, img_name )
        
        img = Image.open(img_path) # æ‰“å¼€å›¾ç‰‡
        label = self.label

        return img, label # è¿”å›æ•°æ®ä¸æ ‡ç­¾
    
    def __len__(self):
        return len(self.imglist)
