from torch.utils.data import Dataset
from PIL import Image
import os

class MyDataet(Dataset):

    def __init__(self, root_dir, label):
        
        # Ëá™ÂÆö‰πâ ÊÄé‰πàÊñπ‰æøÊÄé‰πàÊù•
        self.root_dir = root_dir # ËÆ∞ÂΩïÊï∞ÊçÆÂú∞ÂùÄ‰ª•ÂèäÂØπÂ∫îÁöÑÊ†áÁ≠æ
        self.label = label
        self.imglist = os.listdir(self.root_dir) # ‰ª•ÂàóË°®ÂΩ¢ÂºèÂ±ïÁ§∫Êñá‰ª∂Â§πÂÜÖÊâÄÊúâÊñá‰ª∂
    
    def __getitem__(self, idx):
        img_name= self.imglist[idx]
        img_path = os.path.join( self.root_dir, img_name )
        
        img = Image.open(img_path) # ÊâìÂºÄÂõæÁâá
        label = self.label

        return img, label # ËøîÂõûÊï∞ÊçÆ‰∏éÊ†áÁ≠æ
    
    def __len__(self):
        return len(self.imglist)

        self.sequential = nn.Sequential(
            nn.Conv2d(in_channels=3, out_channels=32, kernel_size=5, padding=2),
            nn.MaxPool2d(2),
            nn.Conv2d(in_channels=32, out_channels=32, kernel_size=5, padding=2),

            nn.Linear(1024, 64),
            nn.Linear(64, 10)

        return x
