def timing(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        import time
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        print(f"Function {func.__name__} took {end_time - start_time:.4f} seconds")
        return result
    return wrapper


@timing
def test(*args, **kwargs):
    print('function test')
test(1,2,3,4,5,6,7,8,9)
